61  Eigen::Ref<Vector<FP>> advance(FP tmax)
62  {
63  // the derivfunktion (i.e. the lambda passed to m_integrator.advance below) requires that there are at least
64  // as many entries in m_flow_result as in Base::m_result
65  assert(m_flow_result.get_num_time_points() == this->get_result().get_num_time_points());
66  auto result = this->get_ode_integrator().advance(
67  [this](auto&& flows, auto&& t, auto&& dflows_dt) {
68  const auto& pop_result = this->get_result();
69  const auto& model = this->get_model();
70  // compute current population
71  // flows contains the accumulated outflows of each compartment for each target compartment at time t.
72  // Using that the ODEs are linear expressions of the flows, get_derivatives can compute the total change
73  // in population from t0 to t.
74  // To incorporate external changes to the last values of pop_result (e.g. by applying mobility), we only
75  // calculate the change in population starting from the last available time point in m_result, instead
76  // of starting at t0. To do that, the following difference of flows is used.
77  model.get_derivatives(flows - m_flow_result.get_value(pop_result.get_num_time_points() - 1),
78  m_pop); // note: overwrites values in pop
79  // add the "initial" value of the ODEs (using last available time point in pop_result)
80  // If no changes were made to the last value in m_result outside of FlowSimulation, the following
81  // line computes the same as `model.get_derivatives(flows, x); x += model.get_initial_values();`.
82  m_pop += pop_result.get_last_value();
83  // compute the current change in flows with respect to the current population
84  dflows_dt.setZero();
85  model.get_flows(m_pop, m_pop, t, dflows_dt); // this result is used by the integrator
86  },
87  tmax, this->get_dt(), m_flow_result);
88  compute_population_results();
89  return result;
90  }

120  void compute_population_results()
121  {
122  const auto& flows = get_flows();
123  const auto& model = this->get_model();
124  auto& result = this->get_result();
125  // take the last time point as base result (instead of the initial results), so that we use external changes
126  const size_t last_tp = result.get_num_time_points() - 1;
127  // calculate new time points
128  for (Eigen::Index i = result.get_num_time_points(); i < flows.get_num_time_points(); i++) {
129  result.add_time_point(flows.get_time(i));
130  model.get_derivatives(flows.get_value(i) - flows.get_value(last_tp), result.get_value(i));
131  result.get_value(i) += result.get_value(last_tp);
132  }
133  }
