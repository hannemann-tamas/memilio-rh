78  TimeSeries<double> const& simulate(int t_max)
79  {
80 
81  m_l = (int)std::floor(parameters.template get<LatencyTime>() / m_dt);
82  m_k =
83  (int)std::ceil((parameters.template get<InfectiousTime>() + parameters.template get<LatencyTime>()) / m_dt);
84  if (m_result.get_time(0) > -(m_k - 1) * m_dt) {
85  log_warning("Constraint check: Initial data starts later than necessary. The simulation may be distorted. "
86  "Start the data at time {:.4f} at the latest.",
87  -(m_k - 1) * m_dt);
88  }
89 
90  while (m_result.get_last_time() < t_max) {
91  m_result.add_time_point(m_result.get_last_time() + m_dt);
92  Eigen::Index idx = m_result.get_num_time_points();
93 
94  // R0t is the effective reproduction number at time t
95  auto R0t1 = parameters.template get<ContactFrequency<double>>().get_cont_freq_mat().get_matrix_at(
96  m_result.get_time(idx - 2))(0, 0) *
97  parameters.template get<TransmissionRisk>() * parameters.template get<InfectiousTime>();
98  auto R0t2 = parameters.template get<ContactFrequency<double>>().get_cont_freq_mat().get_matrix_at(
99  m_result.get_last_time())(0, 0) *
100  parameters.template get<TransmissionRisk>() * parameters.template get<InfectiousTime>();
101 
102  m_result.get_last_value() =
103  Vec::Constant(1, m_result[idx - 2][0] * exp(m_dt * (0.5 * 1 / m_N) *
104  (R0t1 * num_integration_inner_integral(idx - 2) +
105  R0t2 * num_integration_inner_integral(idx - 1))));
106  }
107  return m_result;
108  }

119  TimeSeries<double> const& calculate_EIR()
120  {
121  Eigen::Index num_points = m_result.get_num_time_points();
122  double S, E, I, R;
123  for (Eigen::Index i = m_k; i < num_points; ++i) {
124  S = m_result[i][Eigen::Index(InfectionState::S)];
125  E = m_result[i - m_l][Eigen::Index(InfectionState::S)] - S;
126  I = m_result[i - m_k][Eigen::Index(InfectionState::S)] - m_result[i - m_l][Eigen::Index(InfectionState::S)];
127  R = m_N - S - E - I;
128  m_result_SEIR.add_time_point(m_result.get_time(i), (Vec(4) << S, E, I, R).finished());
129  }
130  return m_result_SEIR;
131  }

145  double generalized_beta_distribution(double tau, double p = 3.0, double q = 10.0) const
146  {
147  if ((parameters.template get<LatencyTime>() < tau) &&
148  (parameters.template get<InfectiousTime>() + parameters.template get<LatencyTime>() > tau)) {
149  return tgamma(p + q) * pow(tau - parameters.template get<LatencyTime>(), p - 1) *
150  pow(parameters.template get<InfectiousTime>() + parameters.template get<LatencyTime>() - tau,
151  q - 1) /
152  (tgamma(p) * tgamma(q) * pow(parameters.template get<InfectiousTime>(), p + q - 1));
153  }
154  return 0.0;
155  }

166  double central_difference_quotient(TimeSeries<double> const& ts_ide, InfectionState compartment,
167  Eigen::Index idx) const
168  {
169  return (ts_ide[idx + 1][Eigen::Index(compartment)] - ts_ide[idx - 1][Eigen::Index(compartment)]) / (2 * m_dt);

179  double num_integration_inner_integral(Eigen::Index idx) const
180  {
181  double res = 0.5 * (generalized_beta_distribution(m_result.get_time(idx) - m_result.get_time(idx - m_k)) *
182  central_difference_quotient(m_result, InfectionState::S, m_k) +
183  generalized_beta_distribution(m_result.get_time(idx) - m_result.get_time(idx - m_l)) *
184  central_difference_quotient(m_result, InfectionState::S, idx - m_l));
185  Eigen::Index i = idx - m_k + 1;
186  while (i <= idx - m_l - 2) {
187  res += (generalized_beta_distribution(m_result.get_time(idx) - m_result.get_time(i)) *
188  central_difference_quotient(m_result, InfectionState::S, i));
189  ++i;
190  }
191  return res;
192  }
