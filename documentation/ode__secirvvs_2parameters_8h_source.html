675  bool apply_constraints()
676  {
677  int corrected = false;
678  if (this->template get<Seasonality<FP>>() < 0.0 || this->template get<Seasonality<FP>>() > 0.5) {
679  log_warning("Constraint check: Parameter Seasonality changed from {} to {}",
680  this->template get<Seasonality<FP>>(), 0);
681  this->template set<Seasonality<FP>>(0);
682  corrected = true;
683  }
684 
685  if (this->template get<ICUCapacity<FP>>() < 0.0) {
686  log_warning("Constraint check: Parameter ICUCapacity changed from {} to {}",
687  this->template get<ICUCapacity<FP>>(), 0);
688  this->template set<ICUCapacity<FP>>(0);
689  corrected = true;
690  }
691 
692  const double tol_times = 1e-1; // accepted tolerance for compartment stays
693 
694  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
695  if (this->template get<IncubationTime<FP>>()[i] < 2 * tol_times) {
696  log_warning("Constraint check: Parameter IncubationTime changed from {} to {}",
697  this->template get<IncubationTime<FP>>()[i], tol_times);
698  this->template get<IncubationTime<FP>>()[i] = 2 * tol_times;
699  corrected = true;
700  }
701 
702  if (2 * this->template get<SerialInterval<FP>>()[i] <
703  this->template get<IncubationTime<FP>>()[i] + tol_times) {
704  log_warning("Constraint check: Parameter SerialInterval changed from {} to {}",
705  this->template get<SerialInterval<FP>>()[i],
706  0.5 * this->template get<IncubationTime<FP>>()[i] + tol_times / 2);
707  this->template get<SerialInterval<FP>>()[i] =
708  0.5 * this->template get<IncubationTime<FP>>()[i] + tol_times / 2;
709  corrected = true;
710  }
711  else if (this->template get<SerialInterval<FP>>()[i] >
712  this->template get<IncubationTime<FP>>()[i] - tol_times / 2) {
713  log_warning("Constraint check: Parameter SerialInterval changed from {} to {}",
714  this->template get<SerialInterval<FP>>()[i],
715  this->template get<IncubationTime<FP>>()[i] - tol_times / 2);
716  this->template get<SerialInterval<FP>>()[i] =
717  this->template get<IncubationTime<FP>>()[i] - tol_times / 2;
