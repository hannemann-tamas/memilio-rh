256 template <class DampingExpression, class DampingSamplings, class F>
257 void apply_dampings(DampingExpression& damping_expression, const DampingSamplings& dampings, F make_matrix)
258 {
259  damping_expression.set_automatic_cache_update(false);
260  for (auto& d : dampings) {
261  for (auto& i : d.get_matrix_indices()) {
262  auto m = make_matrix(double(d.get_value()) * d.get_group_weights());
263  damping_expression[i].add_damping(m, d.get_level(), d.get_type(), d.get_time());
264  }
265  }
266  damping_expression.set_automatic_cache_update(true);
267 }

282 template <class V>
283 auto make_contact_damping_matrix(V&& groups)
284 {
285  auto ones_v = std::decay_t<V>::PlainObject::Constant(groups.size(), 1.0);
286  auto prod = (ones_v - groups) * (ones_v.transpose() - groups.transpose());
287  auto ones_m = decltype(prod)::PlainObject::Constant(groups.size(), groups.size(), 1.0);
288  return ones_m - sqrt(prod.array()).matrix();
289 }

299 template <class V>
300 auto make_migration_damping_vector(ColumnVectorShape shape, V&& groups)
301 {
302  return Eigen::VectorXd::NullaryExpr(shape.size(), [shape, groups = std::forward<V>(groups)](Eigen::Index i) {
303  auto num_groups = groups.size();
304  auto num_compartments = size_t(shape.size()) / num_groups;
305  return groups[size_t(i) / num_compartments];
306  });
307 }
