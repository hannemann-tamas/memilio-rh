107  void get_flows(Eigen::Ref<const Vector<FP>> pop, Eigen::Ref<const Vector<FP>> y, FP t,
108  Eigen::Ref<Vector<FP>> flows) const override
109  {
110  auto const& params = this->parameters;
111  AgeGroup n_agegroups = params.get_num_groups();
112 
113  ContactMatrixGroup const& contact_matrix = params.template get<ContactPatterns<FP>>();
114 
115  auto icu_occupancy = 0.0;
116  auto test_and_trace_required = 0.0;
117  for (auto i = AgeGroup(0); i < n_agegroups; ++i) {
118  test_and_trace_required +=
119  (1 - params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i]) /
120  params.template get<TimeInfectedNoSymptoms<FP>>()[i] *
121  (this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsNaive}) +
122  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsPartialImmunity}) +
123  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsImprovedImmunity}) +
124  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsNaiveConfirmed}) +
125  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed}) +
126  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed}));
127  icu_occupancy += this->populations.get_from(pop, {i, InfectionState::InfectedCriticalNaive}) +
128  this->populations.get_from(pop, {i, InfectionState::InfectedCriticalPartialImmunity}) +
129  this->populations.get_from(pop, {i, InfectionState::InfectedCriticalImprovedImmunity});
130  }
131 
132  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
133 
134  size_t SNi = this->populations.get_flat_index({i, InfectionState::SusceptibleNaive});
135  size_t ENi = this->populations.get_flat_index({i, InfectionState::ExposedNaive});
136  size_t INSNi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsNaive});
137  size_t ISyNi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsNaive});
138  size_t ISevNi = this->populations.get_flat_index({i, InfectionState::InfectedSevereNaive});
139  size_t ICrNi = this->populations.get_flat_index({i, InfectionState::InfectedCriticalNaive});
140 
141  size_t INSNCi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsNaiveConfirmed});
142  size_t ISyNCi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsNaiveConfirmed});
143 
144  size_t SPIi = this->populations.get_flat_index({i, InfectionState::SusceptiblePartialImmunity});
145  size_t EPIi = this->populations.get_flat_index({i, InfectionState::ExposedPartialImmunity});
146  size_t INSPIi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsPartialImmunity});
147  size_t ISyPIi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsPartialImmunity});
148  size_t ISevPIi = this->populations.get_flat_index({i, InfectionState::InfectedSeverePartialImmunity});
149  size_t ICrPIi = this->populations.get_flat_index({i, InfectionState::InfectedCriticalPartialImmunity});
150 
151  size_t INSPICi =
152  this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed});
153  size_t ISyPICi =
154  this->populations.get_flat_index({i, InfectionState::InfectedSymptomsPartialImmunityConfirmed});
155 
156  size_t EIIi = this->populations.get_flat_index({i, InfectionState::ExposedImprovedImmunity});
157  size_t INSIIi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsImprovedImmunity});
158  size_t ISyIIi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsImprovedImmunity});
159  size_t ISevIIi = this->populations.get_flat_index({i, InfectionState::InfectedSevereImprovedImmunity});
160  size_t ICrIIi = this->populations.get_flat_index({i, InfectionState::InfectedCriticalImprovedImmunity});
161 
162  size_t INSIICi =
163  this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed});
164  size_t ISyIICi =
165  this->populations.get_flat_index({i, InfectionState::InfectedSymptomsImprovedImmunityConfirmed});
166 
167  size_t SIIi = this->populations.get_flat_index({i, InfectionState::SusceptibleImprovedImmunity});
168 
169  double reducExposedPartialImmunity = params.template get<ReducExposedPartialImmunity<FP>>()[i];
170  double reducExposedImprovedImmunity = params.template get<ReducExposedImprovedImmunity<FP>>()[i];
171  double reducInfectedSymptomsPartialImmunity =
172  params.template get<ReducInfectedSymptomsPartialImmunity<FP>>()[i];
173  double reducInfectedSymptomsImprovedImmunity =
174  params.template get<ReducInfectedSymptomsImprovedImmunity<FP>>()[i];
175  double reducInfectedSevereCriticalDeadPartialImmunity =
176  params.template get<ReducInfectedSevereCriticalDeadPartialImmunity<FP>>()[i];
177  double reducInfectedSevereCriticalDeadImprovedImmunity =
178  params.template get<ReducInfectedSevereCriticalDeadImprovedImmunity<FP>>()[i];
179  double reducTimeInfectedMild = params.template get<ReducTimeInfectedMild<FP>>()[i];
180 
181  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
182  auto riskFromInfectedSymptomatic =
183  smoother_cosine(test_and_trace_required, params.template get<TestAndTraceCapacity<FP>>(),
184  params.template get<TestAndTraceCapacity<FP>>() * 15,
185  params.template get<RiskOfInfectionFromSymptomatic<FP>>()[i],
186  params.template get<MaxRiskOfInfectionFromSymptomatic<FP>>()[i]);
187 
188  auto riskFromInfectedNoSymptoms =
189  smoother_cosine(test_and_trace_required, params.template get<TestAndTraceCapacity<FP>>(),
190  params.template get<TestAndTraceCapacity<FP>>() * 2,
191  params.template get<RelativeTransmissionNoSymptoms<FP>>()[i], 1.0);
192 
193  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
194  size_t SNj = this->populations.get_flat_index({j, InfectionState::SusceptibleNaive});
195  size_t ENj = this->populations.get_flat_index({j, InfectionState::ExposedNaive});
196  size_t INSNj = this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsNaive});
197  size_t ISyNj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsNaive});
198  size_t ISevNj = this->populations.get_flat_index({j, InfectionState::InfectedSevereNaive});
199  size_t ICrNj = this->populations.get_flat_index({j, InfectionState::InfectedCriticalNaive});
200  size_t SIIj = this->populations.get_flat_index({j, InfectionState::SusceptibleImprovedImmunity});
201 
202  size_t INSNCj = this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsNaiveConfirmed});
203  size_t ISyNCj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsNaiveConfirmed});
204 
205  size_t SPIj = this->populations.get_flat_index({j, InfectionState::SusceptiblePartialImmunity});
206  size_t EPIj = this->populations.get_flat_index({j, InfectionState::ExposedPartialImmunity});
207  size_t INSPIj =
208  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsPartialImmunity});
209  size_t ISyPIj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsPartialImmunity});
210  size_t ISevPIj = this->populations.get_flat_index({j, InfectionState::InfectedSeverePartialImmunity});
211  size_t ICrPIj = this->populations.get_flat_index({j, InfectionState::InfectedCriticalPartialImmunity});
212 
213  size_t INSPICj =
214  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed});
215  size_t ISyPICj =
216  this->populations.get_flat_index({j, InfectionState::InfectedSymptomsPartialImmunityConfirmed});
217 
218  size_t EIIj = this->populations.get_flat_index({j, InfectionState::ExposedImprovedImmunity});
219  size_t INSIIj =
220  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsImprovedImmunity});
221  size_t ISyIIj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsImprovedImmunity});
222  size_t ISevIIj = this->populations.get_flat_index({j, InfectionState::InfectedSevereImprovedImmunity});
223  size_t ICrIIj = this->populations.get_flat_index({j, InfectionState::InfectedCriticalImprovedImmunity});
224 
225  size_t INSIICj =
226  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed});
227  size_t ISyIICj =
228  this->populations.get_flat_index({j, InfectionState::InfectedSymptomsImprovedImmunityConfirmed});
229 
230  // effective contact rate by contact rate between groups i and j and damping j
231  FP season_val =
232  (1 + params.template get<Seasonality<FP>>() *
233  sin(3.141592653589793 * ((params.template get<StartDay>() + t) / 182.5 + 0.5)));
234  FP cont_freq_eff = season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
235  static_cast<Eigen::Index>((size_t)j));
236  // without died people
237  FP Nj = pop[SNj] + pop[ENj] + pop[INSNj] + pop[ISyNj] + pop[ISevNj] + pop[ICrNj] + pop[INSNCj] +
238  pop[ISyNCj] + pop[SPIj] + pop[EPIj] + pop[INSPIj] + pop[ISyPIj] + pop[ISevPIj] + pop[ICrPIj] +
239  pop[INSPICj] + pop[ISyPICj] + pop[SIIj] + pop[EIIj] + pop[INSIIj] + pop[ISyIIj] + pop[ISevIIj] +
240  pop[ICrIIj] + pop[INSIICj] + pop[ISyIICj];
241 
242  FP divNj = 1.0 / Nj; // precompute 1.0/Nj
243 
244  FP ext_inf_force_dummy = cont_freq_eff * divNj *
245  params.template get<TransmissionProbabilityOnContact<FP>>()[(AgeGroup)i] *
246  (riskFromInfectedNoSymptoms * (pop[INSNj] + pop[INSPIj] + pop[INSIIj]) +
247  riskFromInfectedSymptomatic * (pop[ISyNj] + pop[ISyPIj] + pop[ISyIIj]));
248 
249  FP dummy_SN = y[SNi] * ext_inf_force_dummy;
250 
251  FP dummy_SPI = y[SPIi] * reducExposedPartialImmunity * ext_inf_force_dummy;
252 
253  FP dummy_SII = y[SIIi] * reducExposedImprovedImmunity * ext_inf_force_dummy;
254 
255  flows[this->template get_flat_flow_index<InfectionState::SusceptibleNaive,
256  InfectionState::ExposedNaive>({i})] += dummy_SN;
257  flows[this->template get_flat_flow_index<InfectionState::SusceptiblePartialImmunity,
258  InfectionState::ExposedPartialImmunity>({i})] += dummy_SPI;
259  flows[this->template get_flat_flow_index<InfectionState::SusceptibleImprovedImmunity,
260  InfectionState::ExposedImprovedImmunity>({i})] += dummy_SII;
261  }
262 
263  // ICU capacity shortage is close
264  // TODO: if this is used with vaccination model, it has to be adapted if CriticalPerSevere
265  // is different for different vaccination status. This is not the case here and in addition, ICUCapacity
266  // is set to infinity and this functionality is deactivated, so this is OK for the moment.
267  double criticalPerSevereAdjusted = smoother_cosine(
268  icu_occupancy, 0.90 * params.template get<ICUCapacity<FP>>(), params.template get<ICUCapacity<FP>>(),
269  params.template get<CriticalPerSevere<FP>>()[i], 0);
270 
271  FP deathsPerSevereAdjusted = params.template get<CriticalPerSevere<FP>>()[i] - criticalPerSevereAdjusted;
272 
273  /**** path of immune-naive ***/
274  // Exposed
275  flows[this->template get_flat_flow_index<InfectionState::ExposedNaive,
276  InfectionState::InfectedNoSymptomsNaive>({i})] +=
277  y[ENi] / params.template get<TimeExposed<FP>>()[i];
278 
279  // InfectedNoSymptoms
280  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsNaive,
281  InfectionState::SusceptibleImprovedImmunity>({i})] =
282  params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i] /
283  params.template get<TimeInfectedNoSymptoms<FP>>()[i] * y[INSNi];
284  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsNaive,
285  InfectionState::InfectedSymptomsNaive>({i})] =
286  (1 - params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i]) /
287  params.template get<TimeInfectedNoSymptoms<FP>>()[i] * y[INSNi];
288  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsNaiveConfirmed,
289  InfectionState::InfectedSymptomsNaiveConfirmed>({i})] =
290  (1 - params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i]) /
291  params.template get<TimeInfectedNoSymptoms<FP>>()[i] * y[INSNCi];
292  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsNaiveConfirmed,
293  InfectionState::SusceptibleImprovedImmunity>({i})] =
294  params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i] /
295  params.template get<TimeInfectedNoSymptoms<FP>>()[i] * y[INSNCi];
296 
297  // InfectedSymptoms
298  flows[this->template get_flat_flow_index<InfectionState::InfectedSymptomsNaive,
299  InfectionState::InfectedSevereNaive>({i})] =
300  params.template get<SeverePerInfectedSymptoms<FP>>()[i] /
301  params.template get<TimeInfectedSymptoms<FP>>()[i] * y[ISyNi];
302 
303  flows[this->template get_flat_flow_index<InfectionState::InfectedSymptomsNaive,
304  InfectionState::SusceptibleImprovedImmunity>({i})] =
305  (1 - params.template get<SeverePerInfectedSymptoms<FP>>()[i]) /
306  params.template get<TimeInfectedSymptoms<FP>>()[i] * y[ISyNi];
307 
308  flows[this->template get_flat_flow_index<InfectionState::InfectedSymptomsNaiveConfirmed,
309  InfectionState::InfectedSevereNaive>({i})] =
310  params.template get<SeverePerInfectedSymptoms<FP>>()[i] /
311  params.template get<TimeInfectedSymptoms<FP>>()[i] * y[ISyNCi];
312 
313  flows[this->template get_flat_flow_index<InfectionState::InfectedSymptomsNaiveConfirmed,
314  InfectionState::SusceptibleImprovedImmunity>({i})] =
315  (1 - params.template get<SeverePerInfectedSymptoms<FP>>()[i]) /
316  params.template get<TimeInfectedSymptoms<FP>>()[i] * y[ISyNCi];
317 
318  // InfectedSevere
319  flows[this->template get_flat_flow_index<InfectionState::InfectedSevereNaive,
320  InfectionState::InfectedCriticalNaive>({i})] =
321  criticalPerSevereAdjusted / params.template get<TimeInfectedSevere<FP>>()[i] * y[ISevNi];
322 
323  flows[this->template get_flat_flow_index<InfectionState::InfectedSevereNaive,
324  InfectionState::SusceptibleImprovedImmunity>({i})] =
325  (1 - params.template get<CriticalPerSevere<FP>>()[i]) /
326  params.template get<TimeInfectedSevere<FP>>()[i] * y[ISevNi];
327 
328  flows[this->template get_flat_flow_index<InfectionState::InfectedSevereNaive, InfectionState::DeadNaive>(
329  {i})] = deathsPerSevereAdjusted / params.template get<TimeInfectedSevere<FP>>()[i] * y[ISevNi];
330 
331  // InfectedCritical
332  flows[this->template get_flat_flow_index<InfectionState::InfectedCriticalNaive, InfectionState::DeadNaive>(
333  {i})] = params.template get<DeathsPerCritical<FP>>()[i] /
334  params.template get<TimeInfectedCritical<FP>>()[i] * y[ICrNi];
335 
336  flows[this->template get_flat_flow_index<InfectionState::InfectedCriticalNaive,
337  InfectionState::SusceptibleImprovedImmunity>({i})] =
338  (1 - params.template get<DeathsPerCritical<FP>>()[i]) /
339  params.template get<TimeInfectedCritical<FP>>()[i] * y[ICrNi];
340 
341  // /**** path of partially immune (e.g., one dose of vaccination) ***/
342  // Exposed
343  flows[this->template get_flat_flow_index<InfectionState::ExposedPartialImmunity,
344  InfectionState::InfectedNoSymptomsPartialImmunity>({i})] +=
345  y[EPIi] / params.template get<TimeExposed<FP>>()[i];
346 
347  // InfectedNoSymptoms
348  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsPartialImmunity,
349  InfectionState::SusceptibleImprovedImmunity>({i})] =
350  (1 - (reducInfectedSymptomsPartialImmunity / reducExposedPartialImmunity) *
351  (1 - params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i])) /
352  (params.template get<TimeInfectedNoSymptoms<FP>>()[i] * reducTimeInfectedMild) * y[INSPIi];
353  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsPartialImmunity,
354  InfectionState::InfectedSymptomsPartialImmunity>({i})] =
355  (reducInfectedSymptomsPartialImmunity / reducExposedPartialImmunity) *
356  (1 - params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i]) /
357  (params.template get<TimeInfectedNoSymptoms<FP>>()[i] * reducTimeInfectedMild) * y[INSPIi];
358  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsPartialImmunityConfirmed,
359  InfectionState::InfectedSymptomsPartialImmunityConfirmed>({i})] =
360  (reducInfectedSymptomsPartialImmunity / reducExposedPartialImmunity) *
361  (1 - params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i]) /
362  (params.template get<TimeInfectedNoSymptoms<FP>>()[i] * reducTimeInfectedMild) * y[INSPICi];
363  flows[this->template get_flat_flow_index<InfectionState::InfectedNoSymptomsPartialImmunityConfirmed,
364  InfectionState::SusceptibleImprovedImmunity>({i})] =

366  (1 - params.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i])) /

589  void apply_variant(const double t, const CustomIndexArray<UncertainValue<FP>, AgeGroup> base_infectiousness)
590  {
591  auto start_day = this->get_model().parameters.template get<StartDay>();
592  auto start_day_new_variant = this->get_model().parameters.template get<StartDayNewVariant>();
593 
594  if (start_day + t >= start_day_new_variant - 1e-10) {
595  const double days_variant = t - (start_day_new_variant - start_day);
596  const double share_new_variant = std::min(1.0, 0.01 * pow(2, (1. / 7) * days_variant));
597  const auto num_groups = this->get_model().parameters.get_num_groups();
598  for (auto i = AgeGroup(0); i < num_groups; ++i) {
599  double new_transmission =
600  (1 - share_new_variant) * base_infectiousness[i] +
601  share_new_variant * base_infectiousness[i] *
602  this->get_model().parameters.template get<InfectiousnessNewVariant<FP>>()[i];
603  this->get_model().parameters.template get<TransmissionProbabilityOnContact<FP>>()[i] = new_transmission;
604  }
605  }

608  void apply_vaccination(double t)
609  {
610  auto t_idx = SimulationDay((size_t)t);
611  auto& params = this->get_model().parameters;
612  size_t num_groups = (size_t)params.get_num_groups();
613  auto last_value = this->get_result().get_last_value();
614 
615  auto count = (size_t)InfectionState::Count;
616  auto S = (size_t)InfectionState::SusceptibleNaive;
617  auto SV = (size_t)InfectionState::SusceptiblePartialImmunity;
618  auto R = (size_t)InfectionState::SusceptibleImprovedImmunity;
619 
620  for (size_t i = 0; i < num_groups; ++i) {
621 
622  double first_vacc;
623  double full_vacc;
624  if (t_idx == SimulationDay(0)) {
625  first_vacc = params.template get<DailyFirstVaccination<FP>>()[{(AgeGroup)i, t_idx}];
626  full_vacc = params.template get<DailyFullVaccination<FP>>()[{(AgeGroup)i, t_idx}];
627  }
628  else {
629  first_vacc = params.template get<DailyFirstVaccination<FP>>()[{(AgeGroup)i, t_idx}] -
630  params.template get<DailyFirstVaccination<FP>>()[{(AgeGroup)i, t_idx - SimulationDay(1)}];
631  full_vacc = params.template get<DailyFullVaccination<FP>>()[{(AgeGroup)i, t_idx}] -
632  params.template get<DailyFullVaccination<FP>>()[{(AgeGroup)i, t_idx - SimulationDay(1)}];
633  }
634 
635  if (last_value(count * i + S) - first_vacc < 0) {
636  auto corrected = 0.99 * last_value(count * i + S);
637  log_warning("too many first vaccinated at time {}: setting first_vacc from {} to {}", t, first_vacc,
638  corrected);
639  first_vacc = corrected;
640  }
641 
642  last_value(count * i + S) -= first_vacc;
643  last_value(count * i + SV) += first_vacc;
644 
645  if (last_value(count * i + SV) - full_vacc < 0) {
646  auto corrected = 0.99 * last_value(count * i + SV);
647  log_warning("too many fully vaccinated at time {}: setting full_vacc from {} to {}", t, full_vacc,
648  corrected);
649  full_vacc = corrected;
650  }
651 
652  last_value(count * i + SV) -= full_vacc;
653  last_value(count * i + R) += full_vacc;
654  }
655  }

664  Eigen::Ref<Vector<FP>> advance(FP tmax)
665  {
666  auto& t_end_dyn_npis = this->get_model().parameters.get_end_dynamic_npis();
667  auto& dyn_npis = this->get_model().parameters.template get<DynamicNPIsInfectedSymptoms<FP>>();
668  auto& contact_patterns = this->get_model().parameters.template get<ContactPatterns<FP>>();
669  // const size_t num_groups = (size_t)this->get_model().parameters.get_num_groups();
670 
671  // in the apply_variant function, we adjust the TransmissionProbabilityOnContact parameter. We need to store
672  // the base value to use it in the apply_variant function and also to reset the parameter after the simulation.
673  auto base_infectiousness = this->get_model().parameters.template get<TransmissionProbabilityOnContact<FP>>();
674 
675  double delay_lockdown;
676  auto t = BaseT::get_result().get_last_time();
677  const auto dt = dyn_npis.get_interval().get();
678  while (t < tmax) {
679 
680  auto dt_eff = std::min({dt, tmax - t, m_t_last_npi_check + dt - t});
681  if (dt_eff >= 1.0) {
682  dt_eff = 1.0;
683  }
684 
685  if (t == 0) {
686  //this->apply_vaccination(t); // done in init now?
687  this->apply_variant(t, base_infectiousness);
688  }
689  BaseT::advance(t + dt_eff);
690  if (t + 0.5 + dt_eff - std::floor(t + 0.5) >= 1) {
691  this->apply_vaccination(t + 0.5 + dt_eff);
692  this->apply_variant(t, base_infectiousness);
693  }
694 
695  if (t > 0) {
696  delay_lockdown = 7;
697  }
698  else {
699  delay_lockdown = 0;
700  }
701  t = t + dt_eff;
702 
703  if (dyn_npis.get_thresholds().size() > 0) {
704  if (floating_point_greater_equal(t, m_t_last_npi_check + dt)) {
705  if (t < t_end_dyn_npis) {
706  auto inf_rel = get_infections_relative<FP>(*this, t, this->get_result().get_last_value()) *
707  dyn_npis.get_base_value();
708  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
709  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
710  (exceeded_threshold->first > m_dynamic_npi.first ||
711  t > double(m_dynamic_npi.second))) { //old npi was weaker or is expired
712 
713  auto t_start = SimulationTime(t + delay_lockdown);
714  auto t_end = t_start + SimulationTime(dyn_npis.get_duration());
715  this->get_model().parameters.get_start_commuter_detection() = (double)t_start;
716  this->get_model().parameters.get_end_commuter_detection() = (double)t_end;
717  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
718  implement_dynamic_npis(contact_patterns.get_cont_freq_mat(), exceeded_threshold->second,
719  t_start, t_end, [](auto& g) {
720  return make_contact_damping_matrix(g);
721  });
722  }
723  }
724  m_t_last_npi_check = t;
725  }
726  }
727  else {
728  m_t_last_npi_check = t;
729  }
730  }
731  // reset TransmissionProbabilityOnContact. This is important for the graph simulation where the advance
732  // function is called multiple times for the same model.
733  this->get_model().parameters.template get<TransmissionProbabilityOnContact<FP>>() = base_infectiousness;
734 
735  return this->get_result().get_last_value();
736  }

740  std::pair<double, SimulationTime> m_dynamic_npi = {-std::numeric_limits<double>::max(), SimulationTime(0)};
741 };

812 template <typename FP = double, class Base = mio::Simulation<Model<FP>, FP>>
813 auto get_migration_factors(const Simulation<Base>& sim, FP /*t*/, const Eigen::Ref<const Vector<FP>>& y)
814 
815 {
816  auto& params = sim.get_model().parameters;
817  //parameters as arrays
818  auto& p_asymp = params.template get<RecoveredPerInfectedNoSymptoms<FP>>().array().template cast<FP>();
819  auto& p_inf = params.template get<RiskOfInfectionFromSymptomatic<FP>>().array().template cast<FP>();
820  auto& p_inf_max = params.template get<MaxRiskOfInfectionFromSymptomatic<FP>>().array().template cast<FP>();
821  //slice of InfectedNoSymptoms
822  auto y_INS = slice(y, {Eigen::Index(InfectionState::InfectedNoSymptomsNaive),
823  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)}) +
824  slice(y, {Eigen::Index(InfectionState::InfectedNoSymptomsPartialImmunity),
825  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)}) +
826  slice(y, {Eigen::Index(InfectionState::InfectedNoSymptomsImprovedImmunity),
827  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)});
828 
829  //compute isolation, same as infection risk from main model
830  auto test_and_trace_required =
831  ((1 - p_asymp) / params.template get<TimeInfectedNoSymptoms<FP>>().array().template cast<double>() *
832  y_INS.array())
833  .sum();
834  auto riskFromInfectedSymptomatic =
835  smoother_cosine(test_and_trace_required, double(params.template get<TestAndTraceCapacity<FP>>()),
836  params.template get<TestAndTraceCapacity<FP>>() * 5, p_inf.matrix(), p_inf_max.matrix());
837 
838  //set factor for infected
839  auto factors = Eigen::VectorXd::Ones(y.rows()).eval();
840  slice(factors, {Eigen::Index(InfectionState::InfectedSymptomsNaive), Eigen::Index(size_t(params.get_num_groups())),
841  Eigen::Index(InfectionState::Count)})
842  .array() = riskFromInfectedSymptomatic;

844  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)})
845  .array() = riskFromInfectedSymptomatic;
846  slice(factors, {Eigen::Index(InfectionState::InfectedSymptomsImprovedImmunity),
847  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)})
848  .array() = riskFromInfectedSymptomatic;
849  return factors;
