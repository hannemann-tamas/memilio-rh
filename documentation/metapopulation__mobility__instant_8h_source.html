418 template <typename FP>
419 template <class Sim>
420 void MigrationEdge<FP>::apply_migration(FP t, FP dt, SimulationNode<Sim>& node_from, SimulationNode<Sim>& node_to)
421 {
422  //check dynamic npis
423  if (m_t_last_dynamic_npi_check == -std::numeric_limits<double>::infinity()) {
424  m_t_last_dynamic_npi_check = node_from.get_t0();
425  }
426 
427  auto& dyn_npis = m_parameters.get_dynamic_npis_infected();
428  if (dyn_npis.get_thresholds().size() > 0 &&
429  floating_point_greater_equal(t, m_t_last_dynamic_npi_check + dyn_npis.get_interval().get())) {
430  auto inf_rel = get_infections_relative(node_from, t, node_from.get_last_state()) * dyn_npis.get_base_value();
431  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
432  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
433  (exceeded_threshold->first > m_dynamic_npi.first ||
434  t > double(m_dynamic_npi.second))) { //old NPI was weaker or is expired
435  auto t_end = SimulationTime(t + double(dyn_npis.get_duration()));
436  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
437  implement_dynamic_npis(
438  m_parameters.get_coefficients(), exceeded_threshold->second, SimulationTime(t), t_end, [this](auto& g) {
439  return make_migration_damping_vector(m_parameters.get_coefficients().get_shape(), g);
440  });
441  }
442  m_t_last_dynamic_npi_check = t;
443  }
444 
445  //returns
446  for (Eigen::Index i = m_return_times.get_num_time_points() - 1; i >= 0; --i) {
447  if (m_return_times.get_time(i) <= t) {
448  auto v0 = find_value_reverse(node_to.get_result(), m_migrated.get_time(i), 1e-10, 1e-10);
449  assert(v0 != node_to.get_result().rend() && "unexpected error.");
450  calculate_migration_returns<FP, Sim>(m_migrated[i], node_to.get_simulation(), *v0, m_migrated.get_time(i),
451  dt);
452 
453  //the lower-order return calculation may in rare cases produce negative compartments,
454  //especially at the beginning of the simulation.
455  //fix by subtracting the supernumerous returns from the biggest compartment of the age group.
456  Eigen::VectorXd remaining_after_return = (node_to.get_result().get_last_value() - m_migrated[i]).eval();
457  for (Eigen::Index j = 0; j < node_to.get_result().get_last_value().size(); ++j) {
458  if (remaining_after_return(j) < 0) {
459  auto num_comparts = (Eigen::Index)Sim::Model::Compartments::Count;
460  auto group = Eigen::Index(j / num_comparts);
461  auto compart = j % num_comparts;
462  log(remaining_after_return(j) < -1e-3 ? LogLevel::warn : LogLevel::info,
463  "Underflow during migration returns at time {}, compartment {}, age group {}: {}", t, compart,
464  group, remaining_after_return(j));
465  Eigen::Index max_index;
466  slice(remaining_after_return, {group * num_comparts, num_comparts}).maxCoeff(&max_index);
467  log_info("Transferring to compartment {}", max_index);
468  max_index += group * num_comparts;
469  m_migrated[i](max_index) -= remaining_after_return(j);
470  m_migrated[i](j) += remaining_after_return(j);
471  }
472  }
473  node_from.get_result().get_last_value() += m_migrated[i];
474  node_to.get_result().get_last_value() -= m_migrated[i];
475  m_migrated.remove_time_point(i);
476  m_return_times.remove_time_point(i);
477  }
478  }
479 
480  if (!m_return_migrated && (m_parameters.get_coefficients().get_matrix_at(t).array() > 0.0).any()) {
481  //normal daily migration
482  m_migrated.add_time_point(
483  t, (node_from.get_last_state().array() * m_parameters.get_coefficients().get_matrix_at(t).array() *
484  get_migration_factors(node_from, t, node_from.get_last_state()).array())
485  .matrix());
486  m_return_times.add_time_point(t + dt);
487 
488  test_commuters(node_from, m_migrated.get_last_value(), t);
489 
490  node_to.get_result().get_last_value() += m_migrated.get_last_value();
491  node_from.get_result().get_last_value() -= m_migrated.get_last_value();
492  }
493  m_return_migrated = !m_return_migrated;
494 }
