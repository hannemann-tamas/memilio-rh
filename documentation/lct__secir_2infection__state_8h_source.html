179  bool check_constraints() const
180  {
181  if (!(m_subcompartment_numbers.size() == (int)InfectionStateBase::Count)) {
182  log_error("Vector for number of subcompartments has the wrong size.");
183  return true;
184  }
185  if (!(m_subcompartment_numbers[(int)InfectionStateBase::Susceptible] == 1)) {
186  log_error("Susceptible compartment can not have subcompartments.");
187  return true;
188  }
189  if (!(m_subcompartment_numbers[(int)InfectionStateBase::Recovered] == 1)) {
190  log_error("Recovered compartment can not have subcompartments.");
191  return true;
192  }
193  if (!(m_subcompartment_numbers[(int)InfectionStateBase::Dead] == 1)) {
194  log_error("Dead compartment can not have subcompartments.");
195  return true;
196  }
197  for (int i = 0; i < (int)InfectionStateBase::Count; ++i) {
198  if (m_subcompartment_numbers[i] < 1) {
199  log_error("All compartments should have at least one subcompartment.");
200  return true;
201  }
202  }
203  return false;
204  }
