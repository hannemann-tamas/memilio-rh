191  bool check_constraints(ScalarType dt) const
192  {
193  if (!((int)m_transitions.get_num_elements() == (int)InfectionTransition::Count)) {
194  log_error("A variable given for model construction is not valid. Number of elements in transition vector "
195  "does not match the required number.");
196  return true;
197  }
198 
199  for (int i = 0; i < (int)InfectionState::Count; i++) {
200  if (m_populations[0][i] < 0) {
201  log_error("Initialization failed. Initial values for populations are less than zero.");
202  return true;
203  }
204  }
205 
206  // It may be possible to run the simulation with fewer time points, but this number ensures that it is possible.
207  if (m_transitions.get_num_time_points() < (Eigen::Index)std::ceil(get_global_support_max(dt) / dt)) {
208  log_error(
209  "Initialization failed. Not enough time points for transitions given before start of simulation.");
210  return true;
211  }
212 
213  for (int i = 0; i < m_transitions.get_num_time_points(); i++) {
214  for (int j = 0; j < (int)InfectionTransition::Count; j++) {
215  if (m_transitions[i][j] < 0) {
216  log_error("Initialization failed. One or more initial value for transitions is less than zero.");
217  return true;
218  }
219  }
220  }
221  if (m_transitions.get_last_time() != m_populations.get_last_time()) {
222  log_error("Last time point of TimeSeries for transitions does not match last time point of TimeSeries for "
223  "compartments. Both of these time points have to agree for a sensible simulation.");
224  return true;
225  }
226 
227  if (m_populations.get_num_time_points() != 1) {
228  log_error("The TimeSeries for the compartments contains more than one time point. It is unclear how to "
229  "initialize.");
230  return true;
231  }
232 
233  return parameters.check_constraints();
234  }
