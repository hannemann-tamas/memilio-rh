59  bool check_constraints(ScalarType dt) const
60  {
61  if (!((int)m_transitions.get_num_elements() == (int)InfectionTransition::Count)) {
62  log_error("A variable given for model construction is not valid. Number of elements in transition vector "
63  "does not match the required number.");
64  return true;
65  }
66 
67  ScalarType support_max = std::max(
68  {parameters.get<TransitionDistributions>()[(int)InfectionTransition::ExposedToInfectedNoSymptoms]
69  .get_support_max(dt, m_tol),
70  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms]
71  .get_support_max(dt, m_tol),
72  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedNoSymptomsToRecovered]
73  .get_support_max(dt, m_tol),
74  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedSymptomsToInfectedSevere]
75  .get_support_max(dt, m_tol),
76  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedSymptomsToRecovered]
77  .get_support_max(dt, m_tol),
78  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedSevereToInfectedCritical]
79  .get_support_max(dt, m_tol),
80  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedSevereToRecovered]
81  .get_support_max(dt, m_tol),
82  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedCriticalToDead]
83  .get_support_max(dt, m_tol),
84  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedCriticalToRecovered]
85  .get_support_max(dt, m_tol)});
86 
87  if (m_transitions.get_num_time_points() < (Eigen::Index)std::ceil(support_max / dt)) {
88  log_error(
89  "Initialization failed. Not enough time points for transitions given before start of simulation.");
90  return true;
91  }
92 
93  if (m_transitions.get_last_time() != m_populations.get_last_time()) {
94  log_error("Last time point of TimeSeries for transitions does not match last time point of TimeSeries for "
95  "compartments. Both of these time points have to agree for a sensible simulation.");
96  return true;
97  }
98 
99  if (m_populations.get_num_time_points() != 1) {
100  log_error("The TimeSeries for the compartments contains more than one time point. It is unclear how to "
101  "initialize.");
102  return true;
103  }
104 
105  return parameters.check_constraints();
106  }
