126  void initialize(FP dt)
127  {
128  if (m_total_confirmed_cases > 1e-12) {
129  m_initialization_method = 1;
130  other_compartments_current_timestep(dt);
131 
132  // The scheme of the ODE model for initialization is applied here.
133  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Recovered)] =
134  m_total_confirmed_cases -
135  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSymptoms)] -
136  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSevere)] -
137  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedCritical)] -
138  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Dead)];
139 
140  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Susceptible)] =
141  m_N - m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Exposed)] -
142  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedNoSymptoms)] -
143  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSymptoms)] -
144  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSevere)] -
145  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedCritical)] -
146  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Recovered)] -
147  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Dead)];
148  }
149  else {
150 
151  // compute Susceptibles at time 0 and m_forceofinfection at time -dt as initial values for discretization scheme
152  // use m_forceofinfection at -dt to be consistent with further calculations of S (see compute_susceptibles()),
153  // where also the value of m_forceofinfection for the previous timestep is used
154  update_forceofinfection(dt, true);
155  if (m_forceofinfection > 1e-12) {
156  m_initialization_method = 2;
157  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Susceptible)] =
158  m_transitions.get_last_value()[Eigen::Index(InfectionTransition::SusceptibleToExposed)] /
159  (dt * m_forceofinfection);
160 
161  //calculate other compartment sizes for t=0
162  other_compartments_current_timestep(dt);
163 
164  //R; need an initial value for R, therefore do not calculate via compute_recovered()
165  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Recovered)] =
166  m_N - m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Susceptible)] -
167  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Exposed)] -
168  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedNoSymptoms)] -
169  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSymptoms)] -
170  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSevere)] -
171  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedCritical)] -
172  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Dead)];
173  }
174  else if (m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Susceptible)] > 1e-12) {
175  //take initialized value for Susceptibles if value can't be calculated via the standard formula
176  m_initialization_method = 3;
177  //calculate other compartment sizes for t=0
178  other_compartments_current_timestep(dt);
179 
180  //R; need an initial value for R, therefore do not calculate via compute_recovered()
181  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Recovered)] =
182  m_N - m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Susceptible)] -
183  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Exposed)] -
184  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedNoSymptoms)] -
185  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSymptoms)] -
186  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSevere)] -
187  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedCritical)] -
188  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Dead)];
189  }
190  else if (m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Recovered)] > 1e-12) {
191  //if value for Recovered is initialized and standard method is not applicable, calculate Susceptibles via other compartments
192  //determining other compartment sizes is not dependent of Susceptibles(0), just of the transitions of the past.
193  //calculate other compartment sizes for t=0
194  m_initialization_method = 4;
195  other_compartments_current_timestep(dt);
196 
197  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Susceptible)] =
198  m_N - m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Exposed)] -
199  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedNoSymptoms)] -
200  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSymptoms)] -
201  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedSevere)] -
202  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::InfectedCritical)] -
203  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Recovered)] -
204  m_populations[Eigen::Index(0)][Eigen::Index(InfectionState::Dead)];
205  }
206  else {
207  m_initialization_method = -1;
208  log_error(
209  "Error occured while initializing compartments: Force of infection is evaluated to 0 and neither "
210  "Susceptibles nor Recovered or total confirmed cases for time 0 were set. One of them should be "
211  "larger 0.");
212  }
213  }
214  // compute m_forceofinfection at time 0 needed for further simulation
215  update_forceofinfection(dt);
216  }

225  void compute_susceptibles(ScalarType dt)
226  {
227  Eigen::Index num_time_points = m_populations.get_num_time_points();
228  // using number of susceptibles from previous time step and force of infection from previous time step:
229  // compute current number of susceptibles and store susceptibles in m_populations
230  m_populations.get_last_value()[Eigen::Index(InfectionState::Susceptible)] =

247  void compute_flow(int idx_InfectionTransitions, Eigen::Index idx_IncomingFlow, ScalarType dt)
248  {
249  ScalarType sum = 0;
250  /* In order to satisfy TransitionDistribution(dt*i) = 0 for all i >= k, k is determined by the maximum support of the distribution.
251  Since we are using a backwards difference scheme to compute the derivative, we have that the
252  derivative of TransitionDistribution(dt*i) = 0 for all i >= k+1.
253 
254  Hence calc_time_index goes until std::ceil(support_max/dt) since for std::ceil(support_max/dt)+1 all terms are already zero.
255  This needs to be adjusted if we are changing the finite difference scheme */
256 
257  Eigen::Index calc_time_index = (Eigen::Index)std::ceil(
258  parameters.get<TransitionDistributions>()[idx_InfectionTransitions].get_support_max(dt, m_tol) / dt);
259 
260  Eigen::Index num_time_points = m_transitions.get_num_time_points();
261 
262  for (Eigen::Index i = num_time_points - 1 - calc_time_index; i < num_time_points - 1; i++) {
263  // (num_time_points - 1 - i)* dt is the time, the individuals has already spent in this state.
264 
265  ScalarType state_age = (num_time_points - 1 - i) * dt;
266 
267  // backward difference scheme to approximate first derivative
268  sum += (parameters.get<TransitionDistributions>()[idx_InfectionTransitions].eval(state_age) -
269  parameters.get<TransitionDistributions>()[idx_InfectionTransitions].eval(state_age - dt)) /
270  dt * m_transitions[i + 1][idx_IncomingFlow];
271  }
272 
273  m_transitions.get_last_value()[Eigen::Index(idx_InfectionTransitions)] =
274  (-dt) * parameters.get<TransitionProbabilities>()[idx_InfectionTransitions] * sum;
275  }

284  void flows_current_timestep(ScalarType dt)
285  {
286  // calculate flow from S to E with force of infection from previous time step und susceptibles from current time step
287  m_transitions.get_last_value()[Eigen::Index(InfectionTransition::SusceptibleToExposed)] =
288  dt * m_forceofinfection * m_populations.get_last_value()[Eigen::Index(InfectionState::Susceptible)];
289  // calculate all other flows with compute_flow
290  // flow from E to C
291  compute_flow((int)InfectionTransition::ExposedToInfectedNoSymptoms,
292  Eigen::Index(InfectionTransition::SusceptibleToExposed), dt);
293  // flow from C to I
294  compute_flow((int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms,
295  Eigen::Index(InfectionTransition::ExposedToInfectedNoSymptoms), dt);
296  // flow from C to R
297  compute_flow((int)InfectionTransition::InfectedNoSymptomsToRecovered,
298  Eigen::Index(InfectionTransition::ExposedToInfectedNoSymptoms), dt);
299  // flow from I to H
300  compute_flow((int)InfectionTransition::InfectedSymptomsToInfectedSevere,
301  Eigen::Index(InfectionTransition::InfectedNoSymptomsToInfectedSymptoms), dt);
302  // flow from I to R
303  compute_flow((int)InfectionTransition::InfectedSymptomsToRecovered,
304  Eigen::Index(InfectionTransition::InfectedNoSymptomsToInfectedSymptoms), dt);
305  // flow from H to U
306  compute_flow((int)InfectionTransition::InfectedSevereToInfectedCritical,
307  Eigen::Index(InfectionTransition::InfectedSymptomsToInfectedSevere), dt);
308  // flow from to H to R
309  compute_flow((int)InfectionTransition::InfectedSevereToRecovered,
310  Eigen::Index(InfectionTransition::InfectedSymptomsToInfectedSevere), dt);
311  // flow from U to D
312  compute_flow((int)InfectionTransition::InfectedCriticalToDead,
313  Eigen::Index(InfectionTransition::InfectedSevereToInfectedCritical), dt);
314  // flow from U to R
315  compute_flow((int)InfectionTransition::InfectedCriticalToRecovered,
316  Eigen::Index(InfectionTransition::InfectedSevereToInfectedCritical), dt);
317  }

342  void update_forceofinfection(ScalarType dt, bool initialization = false)
343  {
344  m_forceofinfection = 0;
345 
346  // determine the relevant calculation area = union of the supports of the relevant transition distributions
347  ScalarType calc_time = std::max(
348  {parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms]
349  .get_support_max(dt, m_tol),
350  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedNoSymptomsToRecovered]
351  .get_support_max(dt, m_tol),
352  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedSymptomsToInfectedSevere]
353  .get_support_max(dt, m_tol),
354  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedSymptomsToRecovered]
355  .get_support_max(dt, m_tol)});
356 
357  // corresponding index
358  /* need calc_time_index timesteps in sum,
359  subtract 1 because in the last summand all TransitionDistributions evaluate to 0 (by definition of support_max)*/
360  Eigen::Index calc_time_index = (Eigen::Index)std::ceil(calc_time / dt) - 1;
361 
362  Eigen::Index num_time_points;
363  ScalarType current_time;
364  ScalarType deaths;
365 
366  if (initialization) {
367  // determine m_forceofinfection at time -dt which is the penultimate timepoint in m_transitions
368  num_time_points = m_transitions.get_num_time_points() - 1;
369  current_time = -dt;
370  deaths = m_deaths_before;
371  }
372  else {
373  // determine m_forceofinfection for current last time in m_transitions.
374  num_time_points = m_transitions.get_num_time_points();
375  current_time = m_transitions.get_last_time();
376  deaths = m_populations.get_last_value()[Eigen::Index(InfectionState::Dead)];
377  }
378 
379  for (Eigen::Index i = num_time_points - 1 - calc_time_index; i < num_time_points - 1; i++) {
380 
381  ScalarType state_age = (num_time_points - 1 - i) * dt;
382 
383  m_forceofinfection +=
384  parameters.get<TransmissionProbabilityOnContact>().eval(state_age) *
385  parameters.get<ContactPatterns<FP>>().get_cont_freq_mat().get_matrix_at(current_time)(0, 0) *
386  ((parameters.get<TransitionProbabilities>()[(
387  int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms] *
388  parameters
389  .get<TransitionDistributions>()[(
390  int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms]
391  .eval(state_age) +
392  parameters.get<TransitionProbabilities>()[(int)InfectionTransition::InfectedNoSymptomsToRecovered] *
393  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedNoSymptomsToRecovered]
394  .eval(state_age)) *
395  m_transitions[i + 1][Eigen::Index(InfectionTransition::ExposedToInfectedNoSymptoms)] *
396  parameters.get<RelativeTransmissionNoSymptoms>().eval(state_age) +
397  (parameters
398  .get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSymptomsToInfectedSevere] *
399  parameters
400  .get<TransitionDistributions>()[(int)InfectionTransition::InfectedSymptomsToInfectedSevere]
401  .eval(state_age) +
402  parameters.get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSymptomsToRecovered] *
403  parameters.get<TransitionDistributions>()[(int)InfectionTransition::InfectedSymptomsToRecovered]
404  .eval(state_age)) *
405  m_transitions[i + 1][Eigen::Index(InfectionTransition::InfectedNoSymptomsToInfectedSymptoms)] *
406  parameters.get<RiskOfInfectionFromSymptomatic>().eval(state_age));
407  }
408  m_forceofinfection = 1 / (m_N - deaths) * m_forceofinfection;
409  }

429  void compute_compartment(Eigen::Index idx_InfectionState, Eigen::Index idx_IncomingFlow,
430  int idx_TransitionDistribution1, int idx_TransitionDistribution2, ScalarType dt)
431  {
432  ScalarType sum = 0;
433 
434  // determine relevant calculation area and corresponding index
435  ScalarType calc_time =
436  std::max(parameters.get<TransitionDistributions>()[idx_TransitionDistribution1].get_support_max(dt, m_tol),
437  parameters.get<TransitionDistributions>()[idx_TransitionDistribution2].get_support_max(dt, m_tol));
438 
439  Eigen::Index calc_time_index = (Eigen::Index)std::ceil(calc_time / dt) - 1;
440 
441  Eigen::Index num_time_points = m_transitions.get_num_time_points();
442 
443  for (Eigen::Index i = num_time_points - 1 - calc_time_index; i < num_time_points - 1; i++) {
444 
445  ScalarType state_age = (num_time_points - 1 - i) * dt;
446 
447  sum += (parameters.get<TransitionProbabilities>()[idx_TransitionDistribution1] *
448  parameters.get<TransitionDistributions>()[idx_TransitionDistribution1].eval(state_age) +
449  (1 - parameters.get<TransitionProbabilities>()[idx_TransitionDistribution1]) *
450  parameters.get<TransitionDistributions>()[idx_TransitionDistribution2].eval(state_age)) *
451  m_transitions[i + 1][idx_IncomingFlow];
452  }
453 
454  m_populations.get_last_value()[idx_InfectionState] = sum;
455  }
