212  bool step(const DerivFunction<FP>& f, Eigen::Ref<Eigen::VectorXd const> yt, double& t, double& dt,
213  Eigen::Ref<Eigen::VectorXd> ytp1) const override
214  {
215  assert(0 <= m_dt_min);
216  assert(m_dt_min <= m_dt_max);
217 
218  if (dt < m_dt_min || dt > m_dt_max) {
219  mio::log_warning("IntegratorCore: Restricting given step size dt = {} to [{}, {}].", dt, m_dt_min,
220  m_dt_max);
221  }
222 
223  dt = std::min(dt, m_dt_max);
224 
225  double t_eval; // shifted time for evaluating yt
226  double dt_new; // updated dt
227 
228  bool converged = false; // carry for convergence criterion
229  bool dt_is_invalid = false;
230 
231  if (m_yt_eval.size() != yt.size()) {
232  m_yt_eval.resize(yt.size());
233  m_kt_values.resize(yt.size(), m_tab_final.entries_low.size());
234  }
235 
236  m_yt_eval = yt;
237 
238  while (!converged && !dt_is_invalid) {
239  if (dt < m_dt_min) {
240  dt_is_invalid = true;
241  dt = m_dt_min;
242  }
243  // compute first column of kt, i.e. kt_0 for each y in yt_eval
244  f(m_yt_eval, t, m_kt_values.col(0));
245 
246  for (Eigen::Index i = 1; i < m_kt_values.cols(); i++) {
247  // we first compute k_n1 for each y_j, then k_n2 for each y_j, etc.
248  t_eval = t;
249  t_eval += m_tab.entries[i - 1][0] *
250  dt; // t_eval = t + c_i * h // note: line zero of Butcher tableau not stored in array
251  // use ytp1 as temporary storage for evaluating m_kt_values[i]
252  ytp1 = m_yt_eval;
253  for (Eigen::VectorXd::Index k = 1; k < m_tab.entries[i - 1].size(); k++) {
254  ytp1 += (dt * m_tab.entries[i - 1][k]) * m_kt_values.col(k - 1);
255  }
256  // get the derivatives, i.e., compute kt_i for all y in ytp1: kt_i = f(t_eval, ytp1_low)
257  f(ytp1, t_eval, m_kt_values.col(i));
258  }
259  // calculate low order estimate
260  ytp1 = m_yt_eval;
261  ytp1 += (dt * (m_kt_values * m_tab_final.entries_low));
262  // truncation error estimate: yt_low - yt_high = O(h^(p+1)) where p = order of convergence
263  m_error_estimate = dt * (m_kt_values * (m_tab_final.entries_high - m_tab_final.entries_low)).array().abs();
264  // calculate mixed tolerance
265  m_eps = m_abs_tol + ytp1.array().abs() * m_rel_tol;
266 
267  converged = (m_error_estimate <= m_eps).all(); // convergence criterion
268 
269  if (converged || dt_is_invalid) {
270  // if sufficiently exact, return ytp1, which currently contains the lower order approximation
271  // (higher order is not always higher accuracy)
272  t += dt; // this is the t where ytp1 belongs to
273  }
274  // else: repeat the calculation above (with updated dt)
275 
276  // compute new value for dt
277  // converged implies eps/error_estimate >= 1, so dt will be increased for the next step
278  // hence !converged implies 0 < eps/error_estimate < 1, strictly decreasing dt
279  dt_new = dt * std::pow((m_eps / m_error_estimate).minCoeff(), (1. / (m_tab_final.entries_low.size() - 1)));
280  // safety factor for more conservative step increases,
281  // and to avoid dt_new -> dt for step decreases when |error_estimate - eps| -> 0
282  dt_new *= 0.9;
283  // check if updated dt stays within desired bounds and update dt for next step
284  dt = std::min(dt_new, m_dt_max);
285  }
286  dt = std::max(dt, m_dt_min);
287  // return 'converged' in favor of '!dt_is_invalid', as these values only differ if step sizing failed,
288  // but the step with size dt_min was accepted.
289  return converged;
290  }
