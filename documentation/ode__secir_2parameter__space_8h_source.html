39 /* Sets alls SecirParams parameters normally distributed,
40 * using the current value and a given standard deviation
41 * @param[inout] params SecirParams including contact patterns for alle age groups
42 * @param[in] t0 start time
43 * @param[in] tmax end time
44 * @param[in] dev_rel maximum relative deviation from particular value(s) given in params
45 */
46 template <typename FP = double>
47 void set_params_distributions_normal(Model<FP>& model, double t0, double tmax, double dev_rel)
48 {
49  auto set_distribution = [dev_rel](UncertainValue<FP>& v, double min_val = 0.001) {
50  v.set_distribution(ParameterDistributionNormal(
51  //add add limitsfornonsense big values. Also mscv has a problem with a few doublesso this fixes it
52  std::min(std::max(min_val, (1 - dev_rel * 2.6) * v), 0.1 * std::numeric_limits<double>::max()),
53  std::min(std::max(min_val, (1 + dev_rel * 2.6) * v), 0.5 * std::numeric_limits<double>::max()),
54  std::min(std::max(min_val, double(v)), 0.3 * std::numeric_limits<double>::max()),
55  std::min(std::max(min_val, dev_rel * v), std::numeric_limits<double>::max())));
56  };
57 
58  set_distribution(model.parameters.template get<Seasonality<FP>>(), 0.0);
59  set_distribution(model.parameters.template get<ICUCapacity<FP>>());
60  set_distribution(model.parameters.template get<TestAndTraceCapacity<FP>>());
61 
62  // populations
63  for (auto i = AgeGroup(0); i < model.parameters.get_num_groups(); i++) {
64  for (auto j = Index<InfectionState>(0); j < Index<InfectionState>(InfectionState::Count); j++) {
65 
66  // don't touch S and D
67  if (j == InfectionState::Susceptible || j == InfectionState::Dead) {
68  continue;
69  }
70 
71  // variably sized groups
72  set_distribution(model.populations[{i, j}], 0.0);
73  }
74  }
75 
76  // times
77  for (auto i = AgeGroup(0); i < model.parameters.get_num_groups(); i++) {
78 
79  set_distribution(model.parameters.template get<IncubationTime<FP>>()[i]);
80  set_distribution(model.parameters.template get<SerialInterval<FP>>()[i]);
81  set_distribution(model.parameters.template get<TimeInfectedSymptoms<FP>>()[i]);
82  set_distribution(model.parameters.template get<TimeInfectedSevere<FP>>()[i]);
83  set_distribution(model.parameters.template get<TimeInfectedCritical<FP>>()[i]);
84 
85  set_distribution(model.parameters.template get<TransmissionProbabilityOnContact<FP>>()[i]);
86  set_distribution(model.parameters.template get<RelativeTransmissionNoSymptoms<FP>>()[i]);
87  set_distribution(model.parameters.template get<RecoveredPerInfectedNoSymptoms<FP>>()[i]);
88  set_distribution(model.parameters.template get<RiskOfInfectionFromSymptomatic<FP>>()[i]);
89  set_distribution(model.parameters.template get<MaxRiskOfInfectionFromSymptomatic<FP>>()[i]);
90  set_distribution(model.parameters.template get<DeathsPerCritical<FP>>()[i]);
91  set_distribution(model.parameters.template get<SeverePerInfectedSymptoms<FP>>()[i]);
92  set_distribution(model.parameters.template get<CriticalPerSevere<FP>>()[i]);
93  }
94 
95  // dampings
96  auto matrices = std::vector<size_t>();
97  for (size_t i = 0; i < model.parameters.template get<ContactPatterns<FP>>().get_cont_freq_mat().get_num_matrices();
98  ++i) {
99  matrices.push_back(i);
100  }
101  auto groups = Eigen::VectorXd::Constant(Eigen::Index(model.parameters.get_num_groups().get()), 1.0);
102  model.parameters.template get<ContactPatterns<FP>>().get_dampings().emplace_back(
103  mio::UncertainValue<FP>(0.5), mio::DampingLevel(0), mio::DampingType(0),
104  mio::SimulationTime(t0 + (tmax - t0) / 2), matrices, groups);
105  set_distribution(model.parameters.template get<ContactPatterns<FP>>().get_dampings()[0].get_value(), 0.0);
106 }

112 template <typename FP = double>
113 void draw_sample_demographics(Model<FP>& model)
114 {
115  model.parameters.template get<ICUCapacity<FP>>().draw_sample();
116  model.parameters.template get<TestAndTraceCapacity<FP>>().draw_sample();
117 
118  for (auto i = AgeGroup(0); i < model.parameters.get_num_groups(); i++) {
119  double group_total = model.populations.get_group_total(i);
120 
121  model.populations[{i, InfectionState::Exposed}].draw_sample();
122  model.populations[{i, InfectionState::InfectedNoSymptoms}].draw_sample();
123  model.populations[{i, InfectionState::InfectedSymptoms}].draw_sample();
124  model.populations[{i, InfectionState::InfectedSevere}].draw_sample();
125  model.populations[{i, InfectionState::InfectedCritical}].draw_sample();
126  model.populations[{i, InfectionState::Recovered}].draw_sample();
127 
128  // no sampling for dead and total numbers
129  // [...]
130 
131  model.populations.template set_difference_from_group_total<AgeGroup>({i, InfectionState::Susceptible},
132  group_total);
133  model.populations.template set_difference_from_group_total<AgeGroup>({i, InfectionState::Susceptible},
134  model.populations.get_group_total(i));
135  }

194 template <typename FP = double>
195 Graph<Model<FP>, MigrationParameters<FP>> draw_sample(Graph<Model<FP>, MigrationParameters<FP>>& graph)
196 {
197  Graph<Model<FP>, MigrationParameters<FP>> sampled_graph;
198 
199  //sample global parameters
200  auto& shared_params_model = graph.nodes()[0].property;
201  draw_sample_infection(shared_params_model);
202  auto& shared_contacts = shared_params_model.parameters.template get<ContactPatterns<FP>>();
203  shared_contacts.draw_sample_dampings();
204  auto& shared_dynamic_npis = shared_params_model.parameters.template get<DynamicNPIsInfectedSymptoms<FP>>();
205  shared_dynamic_npis.draw_sample();
206 
207  for (auto& params_node : graph.nodes()) {
208  auto& node_model = params_node.property;
209 
210  //sample local parameters
211  draw_sample_demographics(params_node.property);
212 
213  //copy global parameters
214  //save demographic parameters so they aren't overwritten
215  auto local_icu_capacity = node_model.parameters.template get<ICUCapacity<FP>>();
216  auto local_tnt_capacity = node_model.parameters.template get<TestAndTraceCapacity<FP>>();
217  auto local_holidays = node_model.parameters.template get<ContactPatterns<FP>>().get_school_holidays();
218  node_model.parameters = shared_params_model.parameters;
219  node_model.parameters.template get<ICUCapacity<FP>>() = local_icu_capacity;
220  node_model.parameters.template get<TestAndTraceCapacity<FP>>() = local_tnt_capacity;
221  node_model.parameters.template get<ContactPatterns<FP>>().get_school_holidays() = local_holidays;
222 
223  node_model.parameters.template get<ContactPatterns<FP>>().make_matrix();
224  node_model.apply_constraints();
225 
226  sampled_graph.add_node(params_node.id, node_model);
227  }
228 
229  for (auto& edge : graph.edges()) {
230  auto edge_params = edge.property;
231  apply_dampings(edge_params.get_coefficients(), shared_contacts.get_dampings(), [&edge_params](auto& v) {
232  return make_migration_damping_vector(edge_params.get_coefficients().get_shape(), v);
233  });
234  edge_params.set_dynamic_npis_infected(shared_dynamic_npis);
235  sampled_graph.add_edge(edge.start_node_idx, edge.end_node_idx, edge_params);
236  }
237 
238  return sampled_graph;
239 }
