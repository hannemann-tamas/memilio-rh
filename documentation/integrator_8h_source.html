88  Eigen::Ref<Eigen::Matrix<FP, Eigen::Dynamic, 1>> advance(const DerivFunction<FP>& f, const FP tmax, FP& dt,
89  TimeSeries<FP>& results)
90  {
91  using std::fabs;
92  using std::min;
93 
94  const auto t0 = results.get_last_time();
95  assert(tmax > t0);
96  assert(dt > 0);
97 
98  const size_t nb_steps = (int)(ceil((tmax - t0) / dt)); // estimated number of time steps (if equidistant)
99 
100  results.reserve(results.get_num_time_points() + nb_steps);
101 
102  bool step_okay = true;
103 
104  auto t = t0;
105  size_t i = results.get_num_time_points() - 1;
106  while (fabs((tmax - t) / (tmax - t0)) > 1e-10) {
107  //we don't make timesteps too small as the error estimator of an adaptive integrator
108  //may not be able to handle it. this is very conservative and maybe unnecessary,
109  //but also unlikely to happen. may need to be reevaluated
110 
111  auto dt_eff = min(dt, tmax - t);
112  results.add_time_point();
113  step_okay &= m_core->step(f, results[i], t, dt_eff, results[i + 1]);
114  results.get_last_time() = t;
115 
116  ++i;
117 
118  if (fabs((tmax - t) / (tmax - t0)) > 1e-10 || dt_eff > dt) {
119  //store dt only if it's not the last step as it is probably smaller than required for tolerances
120  //except if the step function returns a bigger step size so as to not lose efficiency
121  dt = dt_eff;
122  }
123  }
124 
125  if (!step_okay) {
126  log_warning("Adaptive step sizing failed.");
127  }
128  else if (fabs((tmax - t) / (tmax - t0)) > 1e-14) {
129  log_warning("Last time step too small. Could not reach tmax exactly.");
130  }
131  else {
132  log_info("Adaptive step sizing successful to tolerances.");
133  }
134 
135  return results.get_last_value();
136  }
