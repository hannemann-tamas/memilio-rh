98  Eigen::Ref<Vector<FP>> advance(const DerivFunction<FP>& f, const FP tmax, FP& dt, TimeSeries<FP>& results)
99  {
100  // hint at std functions for ADL
101  using std::fabs;
102  using std::max;
103  const FP t0 = results.get_last_time();
104  assert(tmax > t0);
105  assert(dt > 0);
106 
107  const size_t num_steps =
108  static_cast<size_t>(ceil((tmax - t0) / dt)); // estimated number of time steps (if equidistant)
109 
110  results.reserve(results.get_num_time_points() + num_steps);
111 
112  bool step_okay = true;
113 
114  FP dt_copy; // used to check whether step sizing is adaptive
115  FP dt_restore = 0; // used to restore dt if dt was decreased to reach tmax
116  FP t = t0;
117 
118  for (size_t i = results.get_num_time_points() - 1; fabs((tmax - t) / (tmax - t0)) > 1e-10; ++i) {
119  //we don't make timesteps too small as the error estimator of an adaptive integrator
120  //may not be able to handle it. this is very conservative and maybe unnecessary,
121  //but also unlikely to happen. may need to be reevaluated
122 
123  if (dt > tmax - t) {
124  dt_restore = dt;
125  dt = tmax - t;
126  }
127  dt_copy = dt;
128 
129  results.add_time_point();
130  step_okay &= m_core->step(f, results[i], t, dt, results[i + 1]);
131  results.get_last_time() = t;
132 
133  // if dt has been changed (even slighly) by step, register the current m_core as adaptive
134  m_is_adaptive |= !floating_point_equal(dt, dt_copy);
135  }
136  // if dt was decreased to reach tmax in the last time iteration,
137  // we restore it as it is now probably smaller than required for tolerances
138  dt = max(dt, dt_restore);
139 
140  if (m_is_adaptive) {
141  if (!step_okay) {
142  log_warning("Adaptive step sizing failed. Forcing an integration step of size dt_min.");
143  }
144  else if (fabs((tmax - t) / (tmax - t0)) > 1e-14) {
145  log_warning("Last time step too small. Could not reach tmax exactly.");
146  }
147  else {
148  log_info("Adaptive step sizing successful to tolerances.");
149  }
150  }
151 
152  return results.get_last_value();
153  }
