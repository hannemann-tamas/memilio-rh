178  ContactMatrixGroup make_matrix(bool accum = false)
179  {
180  if (!accum) {
181  m_cont_freq.clear_dampings();
182  }
183 
184  auto make_matrix = [](auto&& v) {
185  return mio::make_contact_damping_matrix(v);
186  };
187  mio::apply_dampings(m_cont_freq, m_dampings, make_matrix);
188 
189  for (auto h : m_school_holidays) {
190  //enable damping at the start of the period
191  auto damping = m_school_holiday_damping;
192  damping.set_time(h.first);
193  mio::apply_dampings(m_cont_freq, make_range(&damping, &damping + 1), make_matrix);
194 
195  //disable damping at the end of the period
196  damping.get_value() = 0.0;
197  damping.set_time(h.second);
198  mio::apply_dampings(m_cont_freq, make_range(&damping, &damping + 1), make_matrix);
199  }
200 
201  return m_cont_freq;
