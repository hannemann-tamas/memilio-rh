61  void get_derivatives(Eigen::Ref<const Vector<FP>> pop, Eigen::Ref<const Vector<FP>> y, FP t,
62  Eigen::Ref<Vector<FP>> dydt) const override
63  {
64  auto params = this->parameters;
65  AgeGroup n_agegroups = params.get_num_groups();
66  ContactMatrixGroup const& contact_matrix = params.template get<ContactPatterns<FP>>();
67 
68  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
69 
70  size_t Si = this->populations.get_flat_index({i, InfectionState::Susceptible});
71  size_t Ii = this->populations.get_flat_index({i, InfectionState::Infected});
72  size_t Ri = this->populations.get_flat_index({i, InfectionState::Recovered});
73 
74  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
75 
76  size_t Sj = this->populations.get_flat_index({j, InfectionState::Susceptible});
77  size_t Ij = this->populations.get_flat_index({j, InfectionState::Infected});
78  size_t Rj = this->populations.get_flat_index({j, InfectionState::Recovered});
79 
80  double Nj = pop[Sj] + pop[Ij] + pop[Rj];
81 
82  double coeffStoI = contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
83  static_cast<Eigen::Index>((size_t)j)) *
84  params.template get<TransmissionProbabilityOnContact<FP>>()[i] / Nj;
85 
86  dydt[Si] += -coeffStoI * y[Si] * pop[Ij];
87  dydt[Ii] += coeffStoI * y[Si] * pop[Ij];
88  }
89  dydt[Ii] -= (1.0 / params.template get<TimeInfected<FP>>()[i]) * y[Ii];
90  dydt[Ri] = (1.0 / params.template get<TimeInfected<FP>>()[i]) * y[Ii];
91  }
92  }
