116  const auto& pop_result = get_result();
117  auto& model = get_model();
118  // compute current population
119  model.get_derivatives(flows - get_flows().get_value(pop_result.get_num_time_points() - 1), m_pop);
120  m_pop += pop_result.get_last_value();
121  // compute the current change in flows with respect to the current population
122  dflows_dt.setZero();
123  model.step_size = get_dt(); // set the current step size
124  model.get_flows(m_pop, m_pop, t, dflows_dt);
