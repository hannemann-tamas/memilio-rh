88  bool step(const mio::DerivFunction<FP>& f, Eigen::Ref<Vector<FP> const> yt, FP& t, FP& dt,
89  Eigen::Ref<Vector<FP>> ytp1) const override
90  {
91  using boost::numeric::odeint::fail;
92  using std::max;
93  assert(0 <= m_dt_min);
94  assert(m_dt_min <= m_dt_max);
95 
96  if (dt < m_dt_min || dt > m_dt_max) {
97  mio::log_warning("IntegratorCore: Restricting given step size dt = {} to [{}, {}].", dt, m_dt_min,
98  m_dt_max);
99  }
100  // set initial values for exit conditions
101  auto step_result = fail;
102  bool is_dt_valid = true;
103  // copy vectors from the references, since the stepper cannot (trivially) handle Eigen::Ref
104  m_ytp1 = ytp1;
105  m_yt = yt;
106  // make a integration step, adapting dt to a possibly larger value on success,
107  // or a strictly smaller value on fail.
108  // stop only on a successful step or a failed step size adaption (w.r.t. the minimal step size m_dt_min)
109  while (step_result == fail && is_dt_valid) {
110  if (dt < m_dt_min) {
111  is_dt_valid = false;
112  dt = m_dt_min;
113  }
114  // we use the scheme try_step(sys, in, t, out, dt) with sys=f, in=y(t_{k}), out=y(t_{k+1}).
115  // this is similiar to do_step, but it can adapt the step size dt. If successful, it also updates t.
116 
117  if constexpr (!is_fsal_stepper) { // prevent compile time errors with fsal steppers
118  step_result = m_stepper.try_step(
119  // reorder arguments of the DerivFunction f for the stepper
120  [&](const Vector<FP>& x, Vector<FP>& dxds, FP s) {
121  dxds.resizeLike(x); // boost resizers cannot resize Eigen::Vector, hence we need to do that here
122  f(x, s, dxds);
123  },
124  m_yt, t, m_ytp1, dt);
125  }
126  }
127  // output the new value by copying it back to the reference
128  ytp1 = m_ytp1;
129  // bound dt from below
130  // the last adaptive step (successful or not) may have calculated a new step size smaller than m_dt_min
131 
132  dt = max(dt, m_dt_min);
133  // check whether the last step failed (which means that m_dt_min was still too large to suffice tolerances)
134  if (step_result == fail) {
135  // adaptive stepping failed, but we still return the result of the last attempt
136  t += m_dt_min;
137  return false;
138  }
139  else {
140  // successfully made an integration step
141  return true;
142  }
143  }
