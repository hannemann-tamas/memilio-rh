77  bool step(const mio::DerivFunction<FP>& f, Eigen::Ref<Eigen::Matrix<FP, Eigen::Dynamic, 1> const> yt, FP& t, FP& dt,
78  Eigen::Ref<Eigen::Matrix<FP, Eigen::Dynamic, 1>> ytp1) const override
79  {
80  // copy y(t) to dydt, to retrieve the VectorXd from the Ref
81  dydt = yt;
82  const FP t_old = t; // t is updated by try_step on a successfull step
83  do {
84  // we use the scheme try_step(sys, inout, t, dt) with sys=f, inout=y(t) for
85  // in-place computation. This is similiar to do_step, but it can update t and dt
86  m_stepper.try_step(
87  // reorder arguments of the DerivFunction f for the stepper
88  [&](const Eigen::Matrix<FP, Eigen::Dynamic, 1>& x, Eigen::Matrix<FP, Eigen::Dynamic, 1>& dxds, FP s) {
89  dxds.resizeLike(x); // try_step calls sys with a vector of size 0 for some reason
90  f(x, s, dxds);
91  },
92  dydt, t, dt);
93  // stop on a successfull step or a failed step size adaption (w.r.t. the minimal step size)
94  } while (t == t_old && dt > m_dt_min);
95  ytp1 = dydt; // output new y(t)
96  return dt > m_dt_min;
97  }
