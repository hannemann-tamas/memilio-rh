255  virtual ScalarType get_support_max(ScalarType dt, ScalarType tol = 1e-10)
256  {
257  ScalarType support_max = 0.;
258 
259  if (!floating_point_equal(m_support_tol, tol, 1e-14) || floating_point_equal(m_support_max, -1., 1e-14)) {
260  while (eval(support_max) >= tol) {
261  support_max += dt;
262  }
263

286  virtual ScalarType get_mean(ScalarType dt = 1., ScalarType tol = 1e-10)
287  {
288  if (!floating_point_equal(m_mean_tol, tol, 1e-14) || floating_point_equal(m_mean, -1., 1e-14)) {
289  // Integration using Trapezoidal rule.
290  ScalarType mean = 0.5 * dt * eval(0 * dt);
291  ScalarType supp_max_idx = std::ceil(get_support_max(dt, tol) / dt);
292  for (int i = 1; i < supp_max_idx; i++) {
293  mean += dt * eval(i * dt);
294  }
295 
296  m_mean = mean;
297  m_mean_tol = tol;
298  }
299  return m_mean;
300  }

705 struct ErlangDensity : public StateAgeFunction {
706 
713  ErlangDensity(unsigned int init_shape, ScalarType init_scale)
714  : StateAgeFunction(init_shape, 0., init_scale)
715  {
716  }
717 
726  ScalarType eval(ScalarType state_age) override
727  {
728  if (state_age < 0) {
729  return 0;
730  }
731  int shape = (int)m_distribution_parameter;
732  return std::pow(state_age / m_scale, shape - 1) / (m_scale * boost::math::factorial<ScalarType>(shape - 1)) *
733  std::exp(-state_age / m_scale);
734  }
735 
745  ScalarType get_support_max(ScalarType dt, ScalarType tol = 1e-10) override
746  {
747  // We are looking for the smallest time value t where function(tau)=0 for all tau>t. Thus support max is bigger
748  // than the mean.
749  // We use, that the density is monotonically decreasing for tau>mean here.
750  ScalarType mean = m_distribution_parameter * m_scale;
751  ScalarType support_max = (ScalarType)dt * (int)(mean / dt);
752 
753  if (!floating_point_equal(m_support_tol, tol, 1e-14) || floating_point_equal(m_support_max, -1., 1e-14)) {
754  while (eval(support_max) >= tol) {
755  support_max += dt;
756  }
757 
758  m_support_max = support_max;
759  m_support_tol = tol;
760  }
761 
762  return m_support_max;
763  }
764 
774  ScalarType get_mean(ScalarType dt = 1., ScalarType tol = 1e-10) override
775  {
776  unused(dt);
777  unused(tol);
778  log_warning("Attention: This function is not suited to be a TransitionDistribution. Do not call in case of "
779  "StateAgeFunctions of type b); see documentation of StateAgeFunction Base class.");
780  return m_distribution_parameter * m_scale;
781  }
