54  void get_derivatives(Eigen::Ref<const Vector<FP>> /* pop */, Eigen::Ref<const Vector<FP>> y, FP /* t */,
55  Eigen::Ref<Vector<FP>> dydt) const override
56  {
57  auto& params = this->parameters;
58 
59  const auto pop_total = y.sum() - y[(size_t)InfectionState::ObjectiveFunction];
60 
61  auto& alpha_a = params.template get<AlphaA<FP>>();
62  auto& alpha_i = params.template get<AlphaI<FP>>();
63  auto& kappa = params.template get<Kappa<FP>>();
64  auto& beta = params.template get<Beta<FP>>();
65  auto& mu = params.template get<Mu<FP>>();
66  auto& t_latent_inverse = params.template get<TLatentInverse<FP>>();
67  auto& rho = params.template get<Rho<FP>>();
68  auto& gamma = params.template get<Gamma<FP>>();
69 
70  const auto& s = y[(size_t)InfectionState::Susceptible];
71  const auto& e = y[(size_t)InfectionState::Exposed];
72  const auto& a = y[(size_t)InfectionState::Asymptomatic];
73  const auto& i = y[(size_t)InfectionState::Infected];
74  const auto& r = y[(size_t)InfectionState::Recovered];
75 
76  dydt[(size_t)InfectionState::Susceptible] =
77  -alpha_a / pop_total * s * a - alpha_i / pop_total * s * i + gamma * r;
78  dydt[(size_t)InfectionState::Exposed] =
79  alpha_a / pop_total * s * a + alpha_i / pop_total * s * i - t_latent_inverse * e;
80  dydt[(size_t)InfectionState::Asymptomatic] = t_latent_inverse * e - kappa * a - rho * a;
81  dydt[(size_t)InfectionState::Infected] = kappa * a - beta * i - mu * i;
82  dydt[(size_t)InfectionState::Recovered] = rho * a + beta * i - gamma * r;
83  dydt[(size_t)InfectionState::Perished] = mu * i;
84  dydt[(size_t)InfectionState::ObjectiveFunction] = 1 - alpha_i - alpha_a + 0.1 * kappa;
85  }
86 };
