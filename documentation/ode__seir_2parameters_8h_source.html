132  int corrected = false;
133  if (this->template get<TimeExposed<FP>>() < tol_times) {

137  this->template get<TimeExposed<FP>>(), tol_times);
138  this->template get<TimeExposed<FP>>() = tol_times;
139  corrected = true;
140  }
141  if (this->template get<TimeInfected<FP>>() < tol_times) {
142  log_warning("Constraint check: Parameter TimeInfected changed from {:.4f} to {:.4f}. Please note that "
143  "unreasonably small compartment stays lead to massively increased run time. Consider to cancel "
144  "and reset parameters.",
145  this->template get<TimeInfected<FP>>(), tol_times);
146  this->template get<TimeInfected<FP>>() = tol_times;
147  corrected = true;
148  }
149  if (this->template get<TransmissionProbabilityOnContact<FP>>() < 0.0 ||
150  this->template get<TransmissionProbabilityOnContact<FP>>() > 1.0) {
151  log_warning("Constraint check: Parameter TransmissionProbabilityOnContact changed from {:0.4f} to {:d} ",
152  this->template get<TransmissionProbabilityOnContact<FP>>(), 0.0);
153  this->template get<TransmissionProbabilityOnContact<FP>>() = 0.0;
154  corrected = true;
155  }

167 
168  if (this->template get<TimeExposed<FP>>() < tol_times) {

172  this->template get<TimeExposed<FP>>(), 0.0);
173  return true;
174  }
175  if (this->template get<TimeInfected<FP>>() < tol_times) {
176  log_error("Constraint check: Parameter TimeInfected {:.4f} smaller or equal {:.4f}. Please note that "
177  "unreasonably small compartment stays lead to massively increased run time. Consider to cancel "
178  "and reset parameters.",
179  this->template get<TimeInfected<FP>>(), 0.0);
180  return true;
181  }
182  if (this->template get<TransmissionProbabilityOnContact<FP>>() < 0.0 ||
183  this->template get<TransmissionProbabilityOnContact<FP>>() > 1.0) {
184  log_error(
185  "Constraint check: Parameter TransmissionProbabilityOnContact {:.4f} smaller {:.4f} or greater {:.4f}",
186  this->template get<TransmissionProbabilityOnContact<FP>>(), 0.0, 1.0);
187  return true;
188  }
189  return false;
