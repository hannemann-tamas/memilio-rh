39 template <typename FP = double>
40 void draw_sample_demographics(Model<FP>& model)
41 {
42  model.parameters.template get<ICUCapacity<FP>>().draw_sample();
43  model.parameters.template get<TestAndTraceCapacity<FP>>().draw_sample();
44 
45  for (auto i = AgeGroup(0); i < model.parameters.get_num_groups(); i++) {
46  double group_total = model.populations.get_group_total(i);
47 
48  //sample initial compartments (with exceptions)
49  for (auto inf_state = Index<InfectionState>(0); inf_state < InfectionState::Count; ++inf_state) {
50  if (inf_state != InfectionState::SusceptibleNaive && //not sampled, fixed after sampling everything else
51  inf_state != InfectionState::DeadNaive && //not sampled, fixed from data
52  inf_state != InfectionState::DeadPartialImmunity && //not sampled, fixed from data
53  inf_state != InfectionState::DeadImprovedImmunity) { //not sampled, fixed from data
54  model.populations[{i, inf_state}].draw_sample();
55  }
56  }
57 
58  //set susceptibles so the total number stays the same as before sampling.
59  //if the new total without susceptibles is already bigger than the previous total
60  //subtract the overflow from SusceptibleImprovedImmunity, susceptibles will then be approximately zero.
61  model.populations[{i, InfectionState::SusceptibleNaive}] = 0;
62  double diff = model.populations.get_group_total(i) - group_total;
63  if (diff > 0) {
64  model.populations[{i, InfectionState::SusceptibleImprovedImmunity}] -= diff;
65  if (model.populations[{i, InfectionState::SusceptibleImprovedImmunity}] < 0.0) {
66  log_error("Negative Compartment after sampling.");
67  }
68  assert(std::abs(group_total - model.populations.get_group_total(i)) < 1e-10 && "Sanity check.");
69  }
70  model.populations.template set_difference_from_group_total<AgeGroup>({i, InfectionState::SusceptibleNaive},
71  group_total);
72  }
73 }

163 template <typename FP = double>
164 Graph<Model<FP>, MigrationParameters<FP>> draw_sample(Graph<Model<FP>, MigrationParameters<FP>>& graph,
165  bool variant_high)
166 {
167  Graph<Model<FP>, MigrationParameters<FP>> sampled_graph;
168 
169  //sample global parameters
170  auto& shared_params_model = graph.nodes()[0].property;
171  draw_sample_infection(shared_params_model);
172  auto& shared_contacts = shared_params_model.parameters.template get<ContactPatterns<FP>>();
173  shared_contacts.draw_sample_dampings();
174  auto& shared_dynamic_npis = shared_params_model.parameters.template get<DynamicNPIsInfectedSymptoms<FP>>();
175  shared_dynamic_npis.draw_sample();
176 
177  double delta_fac;
178  if (variant_high) {
179  delta_fac = 1.6;
180  }
181  else {
182  delta_fac = 1.4;
183  }
184 
185  //infectiousness of virus variants is not sampled independently but depend on base infectiousness
186  for (auto i = AgeGroup(0); i < shared_params_model.parameters.get_num_groups(); ++i) {
187  shared_params_model.parameters.template get<InfectiousnessNewVariant<FP>>()[i] = delta_fac;
188  }
189 
190  for (auto& params_node : graph.nodes()) {
191  auto& node_model = params_node.property;
192 
193  //sample local parameters
194  draw_sample_demographics(params_node.property);
195 
196  //copy global parameters
197  //save demographic parameters so they aren't overwritten
198  auto local_icu_capacity = node_model.parameters.template get<ICUCapacity<FP>>();
199  auto local_tnt_capacity = node_model.parameters.template get<TestAndTraceCapacity<FP>>();
200  auto local_holidays = node_model.parameters.template get<ContactPatterns<FP>>().get_school_holidays();
201  auto local_daily_v1 = node_model.parameters.template get<DailyFirstVaccination<FP>>();
202  auto local_daily_v2 = node_model.parameters.template get<DailyFullVaccination<FP>>();
203  node_model.parameters = shared_params_model.parameters;
204  node_model.parameters.template get<ICUCapacity<FP>>() = local_icu_capacity;
205  node_model.parameters.template get<TestAndTraceCapacity<FP>>() = local_tnt_capacity;
206  node_model.parameters.template get<ContactPatterns<FP>>().get_school_holidays() = local_holidays;
207  node_model.parameters.template get<DailyFirstVaccination<FP>>() = local_daily_v1;
208  node_model.parameters.template get<DailyFullVaccination<FP>>() = local_daily_v2;
209 
210  node_model.parameters.template get<ContactPatterns<FP>>().make_matrix();
211  node_model.apply_constraints();
212 
213  sampled_graph.add_node(params_node.id, node_model);
214  }
215 
216  for (auto& edge : graph.edges()) {
217  auto edge_params = edge.property;
218  //no dynamic NPIs
219  //TODO: add switch to optionally enable dynamic NPIs to edges
220  sampled_graph.add_edge(edge.start_node_idx, edge.end_node_idx, edge_params);
221  }
222 
223  return sampled_graph;
224 }
